        void DriveTreeInit()
        {
            tv.BeginUpdate();
            tv.Nodes.Clear();

            TreeNode n = new TreeNode("C:/mysql", 0, 0);
            tv.Nodes.Add(n);

            GetDirs(n);

            tv.EndUpdate();
        }

        public void GetDirs(TreeNode node)
        {
            DirectoryInfo[] diArray;
            FileInfo[] fiArray;
            node.Nodes.Clear();
            string fullPath = node.FullPath;
            DirectoryInfo di = new DirectoryInfo(fullPath);
            try
            {
                diArray = di.GetDirectories();
                fiArray = di.GetFiles();
            }
            catch
            {
                return;
            }

            foreach (DirectoryInfo dirinfo in diArray)
            {
                TreeNode dir = new TreeNode(dirinfo.Name, 1, 2);
                node.Nodes.Add(dir);

            }
            foreach (FileInfo fileinfo in fiArray)
            {
                TreeNode file = new TreeNode(fileinfo.Name, 1, 1);
                node.Nodes.Add(file);
            }

        }


        private void tv_BeforeExpand(object sender, TreeViewCancelEventArgs e)
        {
            tv.BeginUpdate();

            foreach (TreeNode node in e.Node.Nodes)
            {
                GetDirs(node);
            }

            tv.EndUpdate();
        }
